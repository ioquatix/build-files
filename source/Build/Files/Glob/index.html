def each(&block)
	return to_enum unless block_given?
	
	::Dir.glob(full_pattern, ::File::FNM_DOTMATCH) do |path|
		# Ignore `.` and `..` entries.
		next if path =~ /\/..?$/
		
		yield Path.new(path, @root)
	end
end
